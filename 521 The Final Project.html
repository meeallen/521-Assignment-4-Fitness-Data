def summarize_dataframe(df):
    """
    Generate a summary of each column in a DataFrame, including:
    - Data type
    - Number of missing values
    - Number of unique values
    """
    summary = {
        col: {
            'Type': df[col].dtype,
            'Missing': df[col].isnull().sum(),
            'Unique': df[col].nunique()
        }
        for col in df.columns
    }

    return pd.DataFrame(summary).T

Index by Activity ID

Each of the activities is given its own separate "activity_id", which is now used as the index for our data.

def split_into_activities(df, time_col="timestamp", gap_minutes=10):
    df = df.sort_values(time_col).copy()
    df[time_col] = pd.to_datetime(df[time_col])
    df["time_diff"] = df[time_col].diff()
    activity_gap = pd.Timedelta(minutes=gap_minutes)
    df["new_activity"] = df["time_diff"] > activity_gap
    df["activity_id"] = df["new_activity"].cumsum() + 1
    return df

Clean GPS Data

Turns the raw location data (in Garmins) into latitudinal and longitudinal degrees that we can use and compute with. Converts the "enhanced_altitude" column float64s into integers.

In[12]:
def clean_gps_data(
    df,
    lat_col="position_lat",
    lon_col="position_long",
    alt_col="enhanced_altitude"
):
    """
    Cleans and converts GPS data:
    - Drops rows with missing GPS
    - Converts Garmin semicircles  degrees (float precision)
    - Converts altitude  integer

    Creates new columns:
        position_lat_degrees
        position_long_degrees
        enhanced_alt_int
    """

    df = df.copy()

    # Drop missing GPS coordinates
    df = df.dropna(subset=[lat_col, lon_col])

    # Convert semicircles  degrees (KEEP FLOAT!)
    factor = 180 / (2**31)
    df["position_lat_degrees"] = df[lat_col] * factor
    df["position_long_degrees"] = df[lon_col] * factor

    # Clean altitude
    df["enhanced_alt_int"] = df[alt_col].round().astype(int)

    return df

Find the Great Circle Distance

This function finds the Haversine distance or "great-circle distance", which is calculated using latitude and longitude, and accounts for the curvature of the earth.

In[13]:
def haversine(lat1, lon1, lat2, lon2):
    """
    Computes great-circle distance (meters) between lat/lon points.
    Assumes inputs are in DEGREES.
    """

    R = 6371000  # Earth radius in meters

    lat1, lon1, lat2, lon2 = map(np.radians, [lat1, lon1, lat2, lon2])

    dlat = lat2 - lat1
    dlon = lon2 - lon1

    a = (
        np.sin(dlat / 2)**2
        + np.cos(lat1) * np.cos(lat2) * np.sin(dlon / 2)**2
    )

    c = 2 * np.arcsin(np.sqrt(a))
    return R * c

Summarize Activities in a Dataframe

This function summarizes each activity's duration, distance, altitude change, max heart rate, average speed and activity type. Most of these columns required some data cleaning, usually to put them at a scale and with the right number of decimal places to be easily viewable by the user.

In[105]:
def summarize_activities(
    df,
    time_col="timestamp",
    lat_col="position_lat_degrees",
    lon_col="position_long_degrees",
    alt_col="enhanced_alt_int",
    hr_col="heart_rate"
):
    """
    Builds a summary table with:
        duration (sec)
        distance (meters)
        altitude change
        max heart rate
        average speed
        activity type
    """

    df = df.copy()
    df[time_col] = pd.to_datetime(df[time_col])

    summaries = []

    for activity_id, g in df.groupby("activity_id"):
        g = g.sort_values(time_col)

        # Pairwise points
        lat1 = g[lat_col].shift()
        lon1 = g[lon_col].shift()
        lat2 = g[lat_col]
        lon2 = g[lon_col]

        distances = haversine(lat1, lon1, lat2, lon2)
        total_distance = distances.sum(skipna=True)

        # Duration
        start = g[time_col].iloc[0]
        end = g[time_col].iloc[-1]
        duration = (end - start).total_seconds()

        avg_speed = total_distance / duration if duration > 0 else 0
        activity_type = classify_activity_type(avg_speed)

        summary = {
            "activity_id": activity_id,
            "duration_min": int(duration / 60),
            "distance_km": f"{total_distance / 1000:.1f}",
            "avg_speed_m_s": f"{avg_speed:.1f}",
            "activity_type": activity_type,
            "altitude_change_meters": g[alt_col].max() - g[alt_col].min(),
            "max_heart_rate": g[hr_col].max() if hr_col in g else None
        }

        summaries.append(summary)

    return pd.DataFrame(summaries)

def heart_rate_violin(Distributions, violin_colors, activity_ids, showmeans=False, showmedians=True):
    fig, ax = plt.subplots(figsize=(12, 6))
    
    # Create the violin plot
    violinplt = ax.violinplot(Distributions, showmeans=showmeans, showmedians=showmedians)
    
    # Color the violins
    for i, body in enumerate(violinplt['bodies']):
        body.set_facecolor(violin_colors[i])
        body.set_edgecolor(None)
        body.set_alpha(0.7)
    
    # Customize means and medians
    if showmeans:
        violinplt['cmeans'].set_color('yellow')
    if showmedians:
        violinplt['cmedians'].set_color('black')
    
    # Set x-axis labels
    ax.set_xticks(range(1, len(activity_ids) + 1))
    ax.set_xticklabels(activity_ids, rotation=45)
    ax.set_xlabel('Activity_Id')
    ax.set_ylabel('Heart Rate')
    ax.set_title('Heart Rate Distributions by Activity')
    
    # Add legend for activity_type
    for t, c in color_map.items():
        ax.scatter([], [], color=c, label=t)
    ax.legend(title='Activity Type')
    
    plt.tight_layout()
    plt.show()

import pkgutil
import sys
import importlib

def list_imported_package_versions(print_results=True):
    """
    Finds all currently imported modules that are also installable packages,
    retrieves their versions, and returns a dictionary like:
        {package_name: version}

    Parameters
    ----------
    print_results : bool
        If True, prints the package==version lines.

    Returns
    -------
    dict
        Mapping of package names to version strings.
    """
    modules = {name for _, name, _ in pkgutil.iter_modules()}
    imports = {name for name in sys.modules.keys() if name in modules}

    versions = {}

    for m in sorted(imports):
        try:
            v = importlib.import_module(m).__version__
            versions[m] = v
            if print_results:
                print(f"{m}=={v}")
        except Exception:
            pass

    return versions
